function setActiveTab(clickedTab) {
  // Remover active de todos los tabs en el grupo
  const wrapper = clickedTab.closest('div');
  wrapper.querySelectorAll('.ubits-tab').forEach(tab => {
    tab.classList.remove('ubits-tab--active');
  });
  // Agregar active al tab clickeado
  clickedTab.classList.add('ubits-tab--active');
}
