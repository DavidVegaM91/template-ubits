Estructura del componente

<!-- Avatar con imagen -->
<span class="ubits-avatar ubits-avatar--md">
  <img src="url" alt="Nombre" class="ubits-avatar__img">
</span>

<!-- Avatar con icono (fallback) -->
<span class="ubits-avatar ubits-avatar--md">
  <span class="ubits-avatar__fallback"><i class="far fa-user"></i></span>
</span>

<!-- Profile list (varios avatares +N) -->
<div class="ubits-profile-list ubits-profile-list--md">
  <span class="ubits-profile-list__avatar" style="z-index: 3; margin-right: -5px;">...avatar...</span>
  <span class="ubits-profile-list__count" style="z-index: 0;"><span class="ubits-profile-list__count-text">+2</span></span>
</div>

.ubits-profile-list--xs | --sm | --md | --lg

.ubits-profile-list__count   /* Chip +N */
.ubits-profile-list__count-text

Cómo implementar

Requisitos

CSS: components/avatar.css, general-styles/fontawesome-icons.css (para fallback icono).
JS: components/avatar.js (para renderAvatar y renderProfileList). 
// Avatar único
document.getElementById('mi-avatar').innerHTML = renderAvatar(
  { nombre: 'Patricia Bermúdez', avatar: 'https://...' },
  { size: 'md' }
);

// Profile list
var personas = [
  { nombre: 'Ana García', avatar: 'https://...' },
  { nombre: 'Luis Pérez', avatar: null }
];
document.getElementById('lista').innerHTML = renderProfileList(personas, { size: 'md', maxVisible: 3 });
